    COMMAND ${Slicer_LAUNCH_COMMAND} $<TARGET_FILE:${KIT}CxxTests> ${TESTNAME}
      ${CMAKE_CURRENT_SOURCE_DIR}/${SCENEFILENAME}
    )
endmacro()
