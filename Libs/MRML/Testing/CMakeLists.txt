  add_test(
    NAME vtkMRMLSceneTest2_${SCENEFILENAME}
    COMMAND ${Slicer_LAUNCH_COMMAND} $<TARGET_FILE:${KIT}CxxTests> vtkMRMLSceneTest2
      ${CMAKE_CURRENT_SOURCE_DIR}/${SCENEFILENAME}
    )
endmacro()
